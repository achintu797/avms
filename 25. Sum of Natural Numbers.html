Sum of Natural Numbers Calculator

#include <stdio.h>

int main() {
    int num, sum = 0, i = 1;

    printf("Enter a positive integer: ");
    scanf("%d", &num);

    do {
        sum += i;
        i++;
    } while (i <= num);

    printf("Sum of natural numbers up to %d is: %d\n", num, sum);
    return 0;
}
