Sum Calculator

#include <stdio.h>

int main() {
    int num, sum = 0;

    printf("Enter positive numbers (negative to end): ");
    while (1) {
        scanf("%d", &num);
        if (num < 0) break;
        sum += num;
    }
    printf("Sum is: %d\n", sum);
    return 0;
}
