#include <stdio.h>

int main() {
    int age;
    printf("Enter the age of the student: ");
    scanf("%d", &age);

    if (age >= 5 && age <= 6) {
        printf("Kindergarten\n");
    } else if (age >= 7 && age <= 10) {
        printf("Elementary\n");
    } else if (age >= 11 && age <= 14) {
        printf("Middle School\n");
    } else if (age >= 15 && age <= 18) {
        printf("High School\n");
    } else {
        printf("Not in school age\n");
    }
    return 0;
}
