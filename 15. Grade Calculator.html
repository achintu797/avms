Grade Calculator

#include <stdio.h>

int main() {
    int score;
    char continueCalc;

    do {
        printf("Enter your score (0-100): ");
        scanf("%d", &score);

        switch (score / 10) {
            case 10:
            case 9:
                printf("Grade: A\n");
                break;
            case 8:
                printf("Grade: B\n");
                break;
            case 7:
                printf("Grade: C\n");
                break;
            case 6:
                printf("Grade: D\n");
                break;
            case 5:
            case 4:
            case 3:
            case 2:
            case 1:
            case 0:
                printf("Grade: F\n");
                break;
            default:
                printf("Invalid score! Please enter a score between 0 and 100.\n");
        }

        printf("Do you want to calculate another grade? (y/n): ");
        scanf(" %c", &continueCalc);

    } while (continueCalc == 'y' || continueCalc == 'Y');

    printf("Thank you for using the grade calculator!\n");
    return 0;
}
